To get my UI pieces to include the LyShine functionality I needed I had to edit my build package to include these things:

ly_add_target(
    NAME ${gem_name}.API STATIC
    NAMESPACE Gem
    FILES_CMAKE
        gem_api_files.cmake
        ${pal_dir}/gem_api_files.cmake
    INCLUDE_DIRECTORIES
        PUBLIC
            Include
        PRIVATE
            Source
            ${LY_ROOT_FOLDER}/Gems/LyShine/Code/Source  <----- Actual Directory Inclusion
    BUILD_DEPENDENCIES
        PUBLIC
            AZ::AzCore
            AZ::AzFramework
            Gem::LyShine          <-------- LyShine, and Static
            Gem::LyShine.Static
)
